153

157 // Si la posición actual es menor que la posición objetivo y la dirección es positiva (subiendo)

160 rigid_body->setLinearVelocity(btVector3(0, 2, 0));

162 // Si la posición actual es mayor o igual que la posición objetivo y la dirección es positiva (subiendo)

165 rigid_body->setLinearVelocity(btVector3(0, 0, 0)); // Detiene el movimiento vertical
